\section{مقدمة: لماذا الآن؟}
في ظل الضغوط المتزايدة على المعلمين، ومعدلات "الاحتراق الوظيفي" المرتفعة عالمياً وعربياً، لم تعد ورش العمل التقليدية (Training Workshops) كافية. تشير الدراسات إلى أن ورش العمل التقليدية تنقل فقط \textbf{10\%} من المهارات إلى التطبيق العملي، بينما يرفع الكوتشينج هذه النسبة إلى أكثر من \textbf{80\%} (Joyce \& Showers).

\vspace{1cm}

\begin{quoteBox}
"الكوتشينج ليس إصلاحاً للمعلم السيء، بل هو وسيلة لإطلاق العنان لإمكانات كل معلم، حتى الأكثر تميزاً منهم."
\end{quoteBox}

\section{ما هو الكوتشينج التعليمي؟}
الكوتشينج التعليمي هو علاقة مهنية قائمة على \textbf{الشراكة} (Partnership) بين "الكوتش" والمعلم. لا يمتلك الكوتش السلطة على المعلم (مثل المدير)، بل يمتلك "أدوات الحوار والملاحظة".

\subsection*{المعادلة الأساسية للكوتشينج:}
\[ \text{ملاحظة دقيقة} + \text{بيانات موضوعية} + \text{حوار تأملي} = \text{تحسن مستدام} \]

% --- المبادئ السبعة ---
\section{المبادئ السبعة للكوتشينج الفعال (Jim Knight)}
يقوم الكوتشينج الناجح على مبادئ أساسية تضمن احترام المعلم ومهنيته:

% استخدام شبكة بسيطة لمنع التداخل
\begin{tcbraster}[raster columns=2, raster equal height, raster column skip=5mm, raster row skip=5mm]
    \begin{principleBox}{1. المساواة (Equality)}
    الكوتش والمعلم متساويان في القيمة المهنية، لا يوجد "رئيس" و"مرؤوس" في الجلسة.
    \end{principleBox}
    
    \begin{principleBox}{2. الاختيار (Choice)}
    المعلم هو من يختار الأهداف التي يريد العمل عليها، مما يزيد من التزامه.
    \end{principleBox}
    
    \begin{principleBox}{3. الصوت (Voice)}
    يجب أن يُسمع صوت المعلم وتُحترم آراؤه وأفكاره حول ممارساته الصفية.
    \end{principleBox}
    
    \begin{principleBox}{4. الحوار (Dialogue)}
    العملية هي "تفكير مشترك" وليست إلقاء محاضرات من طرف واحد.
    \end{principleBox}
\end{tcbraster}

\section{دورة الكوتشينج: كيف تعمل؟ (The Impact Cycle)}
تعتمد عملية الكوتشينج على دورة منهجية لضمان النتائج:

\vspace{1cm}
\begin{center}
\begin{tikzpicture}[
    node distance=5cm, % زيادة المسافة بين الدوائر
    phase/.style={circle, draw=brandBlue, fill=white, line width=1mm, minimum size=3.5cm, align=center, text width=2.5cm, drop shadow},
    arrow/.style={->, >=stealth, line width=1mm, color=brandGold, bend left=45}
]
    % العقد (Nodes)
    \node[phase] (identify) {\textbf{1. التحديد} \\ (Identify)};
    \node[phase, below right=of identify] (learn) {\textbf{2. التعلم} \\ (Learn)};
    \node[phase, below left=of identify] (improve) {\textbf{3. التحسين} \\ (Improve)};

    % الأسهم
    \draw[arrow] (identify) to node[auto] {} (learn);
    \draw[arrow] (learn) to node[auto] {} (improve);
    \draw[arrow] (improve) to node[auto] {} (identify);
\end{tikzpicture}
\end{center}

\begin{description}[style=multiline, leftmargin=3.5cm, font=\bfseries\color{brandBlue}]
    \item[1. مرحلة التحديد:] تحديد هدف واضح وواقعي (PEERS Goal) بناءً على الواقع الحالي.
    \item[2. مرحلة التعلم:] تعلم استراتيجية جديدة لتحقيق الهدف عبر النمذجة أو الشرح.
    \item[3. مرحلة التحسين:] التطبيق والقياس والتعديل حتى تحقق النجاح المطلوب.
\end{description}

% --- خرافات وحقائق ---
\section{خرافات شائعة حول الكوتشينج}

\begin{tcbraster}[raster columns=1, raster row skip=10mm] % زيادة المسافة الرأسية
    \begin{mythBox}{الخرافة رقم 1: الكوتشينج للمعلمين الضعفاء فقط}
    \textbf{الحقيقة:} في عالم الرياضة، حتى أبطال الأولمبياد لديهم مدربون. الكوتشينج هو للمحترفين الطموحين الراغبين في التميز.
    \end{mythBox}
    
    \begin{mythBox}{الخرافة رقم 2: الكوتش هو مقيّم سري للإدارة}
    \textbf{الحقيقة:} الكوتشينج يتطلب "جدار حماية" صارم. ما يدور بين الكوتش والمعلم سري تماماً ولا علاقة له بتقييم الأداء السنوي.
    \end{mythBox}
    
    \begin{mythBox}{الخرافة رقم 3: الكوتش يملك كل الإجابات}
    \textbf{الحقيقة:} الكوتش ليس "غوغل". الكوتش خبير في "عملية" مساعدة المعلم على إيجاد الإجابات بنفسه، وليس تلقينه الحلول.
    \end{mythBox}
\end{tcbraster}
